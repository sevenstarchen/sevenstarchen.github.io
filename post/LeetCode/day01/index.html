func findContinuousSequence(target int) [][]int {
   var ret [][]int
   left:=1
   right:=2
   sum:=0
   mid:=target/2+2
   x:=make([]int,mid)
   for i:=0;i< len(x);i++{
      x[i]=i
   }
   for left<=target/2 && left<right {
      for i := left; i <= right; i++ {
         sum += i
      }
      if sum == target {
         ret=append(ret,x[left:right+1])
         left++
      } else if sum < target {
         right++
      } else {
         left++
      }
      sum=0
   }
   return ret
}
